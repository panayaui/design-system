###### Component
```ts
export class ProgressChartComponent {
  @Input() values: {
    riskType: ProgressChartEnum,
    value: number
  }[];
  @Input() label: string;
  @Input() ariaLabel: string;
  @Input() isGeneralProgress: boolean;
  public progressChartEnum = ProgressChartEnum;
}
```

###### ProgressChartEnum
```ts
export enum ProgressChartEnum {
  SUCCESS = 'success',
  LOW = 'low',
  MEDIUM = 'medium',
  HIGH = 'high',
  VERYHIGH = 'veryHigh',
  INCOMPLETE = 'incomplete',
  COMPLETE = 'complete',
}
```
